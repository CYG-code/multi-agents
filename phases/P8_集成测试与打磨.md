import pytest
import asyncio
import websockets
import json

@pytest.mark.asyncio
async def test_multi_user_chat():
    """测试3个用户同时在线聊天"""
    room_id = "test-room-uuid"
    tokens = [get_test_token(f"student{i}") for i in range(3)]
    
    received_messages = [[] for _ in range(3)]
    
    async def user_session(user_idx, token):
        async with websockets.connect(f"ws://localhost:8000/ws/{room_id}") as ws:
            # 鉴权
            await ws.send(json.dumps({"type": "auth", "token": token}))
            
            # 发送消息
            await asyncio.sleep(0.5)
            await ws.send(json.dumps({
                "type": "chat:message",
                "content": f"用户{user_idx}的测试消息",
                "mentions": []
            }))
            
            # 接收5条消息
            for _ in range(5):
                try:
                    msg = await asyncio.wait_for(ws.recv(), timeout=5.0)
                    received_messages[user_idx].append(json.loads(msg))
                except asyncio.TimeoutError:
                    break
    
    await asyncio.gather(*[
        user_session(i, tokens[i]) for i in range(3)
    ])
    
    # 验证每个用户都收到了消息
    for msgs in received_messages:
        chat_msgs = [m for m in msgs if m.get("type") == "chat:new_message"]
        assert len(chat_msgs) > 0

### 步骤 3：Claude API 限流保护

**文件：** `backend/app/agents/claude_client.py`（扩展）

```python
import asyncio
from collections import deque

class RateLimitedClient:
    """
    Claude API 调用限流保护：
    - 最大并发调用数：3（同时进行的 API 请求不超过3个）
    - 指数退避重试（429 错误时）
    - 全局调用速率监控
    """
    
    def __init__(self):
        self._semaphore = asyncio.Semaphore(3)  # 最大并发数
        self._call_times = deque(maxlen=100)    # 最近100次调用时间
        self._client = None
    
    async def create_message(self, **kwargs):
        """带限流保护的消息创建"""
        async with self._semaphore:
            return await self._create_with_retry(**kwargs)
    
    async def _create_with_retry(self, max_retries=3, **kwargs):
        """指数退避重试"""
        for attempt in range(max_retries):
            try:
                client = self._get_client()
                response = await client.messages.create(**kwargs)
                self._call_times.append(time.time())
                return response
            
            except anthropic.RateLimitError as e:
                if attempt == max_retries - 1:
                    raise
                wait_time = (2 ** attempt) + random.uniform(0, 1)
                print(f"[Claude] 429 Too Many Requests，{wait_time:.1f}秒后重试")
                await asyncio.sleep(wait_time)
            
            except anthropic.APITimeoutError:
                if attempt == max_retries - 1:
                    raise
                await asyncio.sleep(2)
    
    def get_call_rate(self) -> float:
        """返回最近1分钟的调用次数"""
        now = time.time()
        recent = [t for t in self._call_times if now - t < 60]
        return len(recent)


rate_limited_client = RateLimitedClient()
```

#### 6.3 API Rate Limiting

**文件：** `backend/app/middleware/rate_limit.py`

```python
from fastapi import Request, HTTPException
import time
from app.db.redis_client import redis_client

async def rate_limit_middleware(request: Request, call_next):
    """
    简单 IP 级别限流：每分钟最多 60 次 API 请求
    """
    client_ip = request.client.host
    key = f"rate_limit:{client_ip}:{int(time.time() // 60)}"
    
    count = await redis_client.incr(key)
    await redis_client.expire(key, 60)
    
    if count > 60:
        raise HTTPException(
            status_code=429,
            detail="请求过于频繁，请稍后再试"
        )
    
    response = await call_next(request)
    return response
```

## 四、Bug 修复优先级清单

P8 阶段按以下优先级排查和修复问题：

### 🔴 P0（阻塞性）
- WebSocket 连接断线后无法重连
- AI 消息流式输出中断（status 卡在 streaming）
- 数据库事务死锁

### 🟠 P1（高优先级）
- Agent 调度循环触发（短时间大量 AI 消息）
- Yjs 文档丢失数据（协同编辑冲突）
- JWT Token 过期后 WebSocket 静默断开无提示

### 🟡 P2（中优先级）
- 消息历史分页滚动时 UI 抖动
- 教师面板图表数据格式异常（null 值处理）
