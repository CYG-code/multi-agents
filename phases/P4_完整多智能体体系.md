import json, time
from app.db.redis_client import redis_client

QUEUE_KEY_PREFIX = "agent_queue"

def queue_key(room_id: str) -> str:
    return f"{QUEUE_KEY_PREFIX}:{room_id}"

async def enqueue_task(room_id: str, task: dict, delay_seconds: float = 0):
    """
    鍐欏叆浠诲姟鍒?Sorted Set 寤惰繜闃熷垪銆?    score = 鏈熸湜鎵ц鏃堕棿鐨?Unix 鏃堕棿鎴炽€?    """
    execute_at = time.time() + delay_seconds
    task_json = json.dumps(task, ensure_ascii=False)
    await redis_client.zadd(queue_key(room_id), {task_json: execute_at})

async def dequeue_tasks(room_id: str) -> list[dict]:
    """
    鎷夊彇褰撳墠鏃堕棿鎴充箣鍓嶏紙score <= now锛夌殑鎵€鏈夊埌鏈熶换鍔°€?    浣跨敤 ZRANGEBYSCORE 鏌ヨ锛屾秷璐规垚鍔熷悗 ZREM 鍒犻櫎銆?    """
    now = time.time()
    raw_tasks = await redis_client.zrangebyscore(
        queue_key(room_id), min=0, max=now
    )
    
    if not raw_tasks:
        return []
    
    # 鍘熷瓙鍒犻櫎锛堝凡鎷夊彇鐨勪换鍔′粠闃熷垪绉婚櫎锛?    await redis_client.zrem(queue_key(room_id), *raw_tasks)
    
    return [json.loads(t) for t in raw_tasks]

async def requeue_task(room_id: str, task: dict, delay_seconds: float = 5):
    """灏嗕换鍔￠噸鏂板叆闃燂紙寤惰繜鎵ц锛岀敤浜庨攣绔炰簤澶辫触鏃剁殑閲嶈瘯锛?""
