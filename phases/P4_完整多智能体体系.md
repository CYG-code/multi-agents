### 步骤 2：实现通用 RoleAgent 基类

**文件：** `backend/app/agents/role_agents.py`（重构 P3 版本）

```python
import uuid, json, time
from abc import ABC
from datetime import datetime
from sqlalchemy import update
from app.agents.claude_client import stream_completion
from app.db.redis_client import redis_client
from app.db.session import AsyncSessionLocal
from app.models.message import Message, SenderType, MessageStatus
from app.services.message_service import MessageService

class BaseRoleAgent(ABC):
    ROLE: str           # 子类必须定义（如 "facilitator"）
    ROLE_DISPLAY_NAME: str
    PROMPT_FILE: str    # 子类定义 prompt 文件名
    MODEL: str = "claude-3-5-sonnet-20241022"
    MAX_TOKENS: int = 512
    
    def __init__(self):
        prompt_path = f"app/agents/prompts/{self.PROMPT_FILE}"
        with open(prompt_path, encoding="utf-8") as f:
            self._prompt_template = f.read()
    
    def build_system_prompt(self, context: dict, task: dict | None = None) -> str:
        """
        构建完整 System Prompt，注入上下文变量。
        task 字段来自 ChiefDispatcher 的调度指令。
        """
        mention_context = ""
        if task and task.get("trigger_type") == "mention":
            student_name = task.get("student_name", "同学")
            mention_context = (
                f"【系统提示】：用户 {student_name} 刚刚在聊天中直接 @ 了你。"
                f"请针对他/她的这条提问给出直接、正面的回应。"
                f"不要长篇大论，保持你作为 {self.ROLE_DISPLAY_NAME} 的角色设定。"
            )
        
        return self._prompt_template.format(
            task_description=context.get("task_description", ""),
            members_info=context.get("members_info", ""),
            current_phase=context.get("current_phase", ""),
            intervention_reason=task.get("reason", "请适时发言") if task else "请适时发言",
            strategy=task.get("strategy", "") if task else "",
            mention_context=mention_context
        )
    
    def build_messages(self, history: list[dict]) -> list[dict]:
        """将历史聊天记录转换为 Claude messages 格式"""
        formatted = [
            {
                "role": "user",
                "content": f"[{msg['display_name']}]: {msg['content']}"
            }
            for msg in history[-30:]
        ]
        formatted.append({
            "role": "user",
            "content": f"（请根据以上讨论，以{self.ROLE_DISPLAY_NAME}身份发言）"
        })
        return formatted
    
    async def generate_and_push(
        self,
        room_id: str,
        context: dict,
        history: list[dict],
        task: dict | None = None
    ):
        """完整的生成+推送流程（与 P3 相同逻辑，提炼为基类）"""
        message_id = str(uuid.uuid4())
        
        # 写入 streaming 状态的消息占位
        async with AsyncSessionLocal() as db:
            seq_num = await MessageService.get_next_seq_num(room_id)
            msg = Message(
                id=message_id,
                room_id=room_id,
                seq_num=seq_num,
                sender_type=SenderType.agent,
                agent_role=self.ROLE,
                content="",
                status=MessageStatus.streaming
            )
            db.add(msg)
            await db.commit()
        
        await self._broadcast(room_id, {
            "type": "agent:typing", "agent_role": self.ROLE, "is_typing": True
        })
        
        full_content = ""
        success = True
        
        try:
            system_prompt = self.build_system_prompt(context, task)
            messages = self.build_messages(history)
            
            async for token in stream_completion(
                system_prompt=system_prompt,
                messages=messages,
                model=self.MODEL,
                max_tokens=self.MAX_TOKENS
            ):
                full_content += token
                await self._broadcast(room_id, {
                    "type": "agent:stream",
                    "agent_role": self.ROLE,
                    "message_id": message_id,
                    "token": token
                })
        except Exception as e:
            print(f"[{self.__class__.__name__}] 生成失败: {e}")
            success = False
        
        # 更新数据库
        async with AsyncSessionLocal() as db:
            await db.execute(
                update(Message)
                .where(Message.id == message_id)
                .values(
                    content=full_content,
                    status=MessageStatus.ok if success else MessageStatus.failed
                )
            )
            await db.commit()
        
        await self._broadcast(room_id, {
            "type": "agent:stream_end",
            "agent_role": self.ROLE,
            "message_id": message_id,
            "status": "ok" if success else "failed",
            "content": full_content,
            "created_at": datetime.utcnow().isoformat()
        })
        
        await self._broadcast(room_id, {
            "type": "agent:typing", "agent_role": self.ROLE, "is_typing": False
        })
    
    async def _broadcast(self, room_id: str, data: dict):
        await redis_client.publish(f"room:{room_id}", json.dumps(data))
```

### 步骤 3：实现 Redis 任务队列（Sorted Set 延迟队列）

**文件：** `backend/app/agents/queue.py`

```python
import json, time
from app.db.redis_client import redis_client

QUEUE_KEY_PREFIX = "agent_queue"

def queue_key(room_id: str) -> str:
    return f"{QUEUE_KEY_PREFIX}:{room_id}"

async def enqueue_task(room_id: str, task: dict, delay_seconds: float = 0):
    """
    写入任务到 Sorted Set 延迟队列。
    score = 期望执行时间的 Unix 时间戳。
    """
    execute_at = time.time() + delay_seconds
    task_json = json.dumps(task, ensure_ascii=False)
    await redis_client.zadd(queue_key(room_id), {task_json: execute_at})

async def dequeue_tasks(room_id: str) -> list[dict]:
    """
    拉取当前时间戳之前（score <= now）的所有到期任务。
    使用 ZRANGEBYSCORE 查询，消费成功后 ZREM 删除。
    """
    now = time.time()
    raw_tasks = await redis_client.zrangebyscore(
        queue_key(room_id), min=0, max=now
    )
    
    if not raw_tasks:
        return []
    
    # 原子删除（已拉取的任务从队列移除）
    await redis_client.zrem(queue_key(room_id), *raw_tasks)
    
    return [json.loads(t) for t in raw_tasks]

async def requeue_task(room_id: str, task: dict, delay_seconds: float = 5):
    """将任务重新入队（延迟执行，用于锁竞争失败时的重试）"""
    await enqueue_task(room_id, task, delay_seconds)
```

### 步骤 4：实现 AgentWorker（队列消费 + 分布式锁）

**文件：** `backend/app/agents/agent_worker.py`

```python
import asyncio, json, uuid
from app.agents.queue import dequeue_tasks, requeue_task
from app.agents.role_agents import ROLE_AGENTS
from app.agents.context_builder import get_room_context, get_recent_messages
from app.db.redis_client import redis_client

WORKER_ID = str(uuid.uuid4())
LOCK_TTL_SECONDS = 30
POLL_INTERVAL_SECONDS = 1  # 每秒轮询一次队列

class AgentWorker:
    def __init__(self):
        self._running = False
    
    async def run(self):
        """后台协程：持续轮询所有活跃房间的任务队列"""
        self._running = True
        print(f"[AgentWorker] 启动，Worker ID: {WORKER_ID}")
        
        while self._running:
            try:
                await self._process_all_rooms()
            except Exception as e:
                print(f"[AgentWorker] 处理异常: {e}")
            
            await asyncio.sleep(POLL_INTERVAL_SECONDS)
    
    def stop(self):
        self._running = False
    
    async def _process_all_rooms(self):
        """扫描所有活跃房间，处理到期任务"""
        active_rooms = await redis_client.smembers("active_rooms")
        
        for room_id in active_rooms:
            tasks = await dequeue_tasks(room_id)
            for task in tasks:
                # 按 priority 排序（priority 越小越优先处理）
                await self._execute_task(room_id, task)
    
    async def _execute_task(self, room_id: str, task: dict):
        """执行单个调度任务（含分布式锁）"""
        agent_role = task.get("agent_role")
        
        if agent_role not in ROLE_AGENTS:
            print(f"[AgentWorker] 未知角色: {agent_role}")
            return
        
        lock_key = f"room:{room_id}:agent_lock"
        
        # 尝试获取分布式锁（NX = Not Exists，原子性）
        acquired = await redis_client.set(
            lock_key, WORKER_ID,
            nx=True,           # 仅在不存在时设置
            ex=LOCK_TTL_SECONDS  # 30秒后自动释放（防死锁）
        )
        
        if not acquired:
            # 锁被占用：将任务重新入队（延迟5秒重试）
            await requeue_task(room_id, task, delay_seconds=5)
            return
        
        try:
            agent = ROLE_AGENTS[agent_role]
            context = await get_room_context(room_id)
            history = await get_recent_messages(room_id)
            
            await agent.generate_and_push(
                room_id=room_id,
                context=context,
                history=history,
                task=task
            )
        finally:
            # 释放锁（验证仍是本 Worker 持有）
            current_lock = await redis_client.get(lock_key)
            if current_lock == WORKER_ID:
                await redis_client.delete(lock_key)


# 全局 Worker 实例
agent_worker = AgentWorker()
```

**在 `main.py` lifespan 中启动 Worker：**

```python
@asynccontextmanager
async def lifespan(app: FastAPI):
    await init_redis()
    start_scheduler()
    
    # 启动 AgentWorker 后台协程
    worker_task = asyncio.create_task(agent_worker.run())
    
    yield
    
    # 优雅停止
    agent_worker.stop()
    worker_task.cancel()
    try:
        await worker_task
    except asyncio.CancelledError:
        pass
    
    stop_scheduler()
    await close_redis()
```

### 步骤 5：实现基础版专家委员会

**文件：** `backend/app/agents/committee.py`（基础版）

P4 阶段实现基础版（三位 Analyst 用简单规则代替 AI 分析，P5 替换为 AI 分析）：

```python
import asyncio, time
from app.agents.queue import enqueue_task
from app.agents.context_builder import get_recent_messages, get_room_members
from app.db.redis_client import redis_client

class BasicCommittee:
    """P4 阶段基础版专家委员会：用统计规则模拟分析，验证调度流程"""
    
    async def analyze_and_dispatch(self, room_id: str):
        """分析当前协作状态，输出调度决策写入 Redis 队列"""
        messages = await get_recent_messages(room_id, limit=50)
        members = await get_room_members(room_id)
        
        if not messages:
            return
        
        # 并发执行三种分析（P4 用统计规则，P5 替换为 AI）
        cognitive, emotional, interaction = await asyncio.gather(
            self._cognitive_analysis(messages),
            self._emotional_analysis(messages),
            self._interaction_analysis(messages, members)
        )
        
        # 基于分析结果生成调度决策
        interventions = self._dispatch(cognitive, emotional, interaction)
        
        for intervention in interventions:
            await enqueue_task(room_id, {
                "room_id": room_id,
                "agent_role": intervention["role"],
                "reason": intervention["reason"],
                "priority": intervention["priority"],
                "triggered_at": time.time()
            })
    
    async def _cognitive_analysis(self, messages: list) -> dict:
        """认知层分析：计算论点多样性（基于唯一说话者数量）"""
        unique_senders = len(set(m.get("sender_id") for m in messages if m.get("sender_id")))
        total = len(messages)
        
        diversity_score = min(unique_senders / max(total * 0.3, 1), 1.0)
        progress_score = min(total / 20, 1.0)  # 简单估算
        
        return {
            "diversity_score": diversity_score,
            "progress_score": progress_score
        }
    
    async def _emotional_analysis(self, messages: list) -> dict:
        """情绪层分析：基于消极词汇检测（简单规则）"""
        negative_keywords = ["算了", "没意思", "随便", "都行", "无聊", "放弃"]
        conflict_keywords = ["你不对", "你错了", "不可能", "怎么可以"]
        
        recent = [m["content"] for m in messages[-10:]]
        text_combined = " ".join(recent)
        
        passive = any(kw in text_combined for kw in negative_keywords)
        conflict = any(kw in text_combined for kw in conflict_keywords)
        
        return {
            "emotion_flags": {
                "passive": passive,
                "conflict": conflict,
                "anxious": False
            }
        }
    
    async def _interaction_analysis(self, messages: list, members: list) -> dict:
        """互动层分析：计算发言均衡性"""
        from collections import Counter
        sender_counts = Counter(
            m.get("sender_id") for m in messages
            if m.get("sender_type") == "student"
        )
        
        if not sender_counts or not members:
            return {"participation_scores": {}, "balance_score": 0.5}
        
        values = list(sender_counts.values())
        total = sum(values)
        
        # 均衡性 = 1 - 最大占比（越均衡越接近1）
        max_ratio = max(values) / total
        balance_score = 1.0 - max_ratio
        
        return {
            "participation_scores": dict(sender_counts),
            "balance_score": balance_score
        }
    
    def _dispatch(self, cognitive: dict, emotional: dict, interaction: dict) -> list:
        """综合三份分析结果，生成调度决策"""
        interventions = []
        
        # B 类触发判断（阈值来自 agent_settings.yaml，P4 暂时硬编码）
        if cognitive["diversity_score"] < 0.3:
            interventions.append({
                "role": "devil_advocate",
                "reason": f"观点多样性偏低({cognitive['diversity_score']:.2f})，需要引入不同声音",
                "priority": 1
            })
        
        if emotional["emotion_flags"]["conflict"]:
            interventions.append({
                "role": "summarizer",
                "reason": "检测到冲突性发言，需要梳理共识",
                "priority": 1
            })
        
        if emotional["emotion_flags"]["passive"] and interaction["balance_score"] < 0.4:
            interventions.append({
                "role": "encourager",
                "reason": "存在消极情绪且协作均衡性低，需要激励参与",
                "priority": 1
            })
        
        return interventions


basic_committee = BasicCommittee()
```

### 步骤 6：实现 A 类事件触发检测器

**文件：** `backend/app/analysis/triggers.py`

```python
import time
from app.agents.queue import enqueue_task
from app.db.redis_client import redis_client

MONOPOLY_THRESHOLD = 5      # 连续5条来自同一人触发激励者

class TriggerDetector:
    
    async def check_monopoly(self, room_id: str, sender_id: str):
        """
        同步规则：单人垄断检测。
        在每条新消息入库后调用。
        """
        # 使用 Redis List 维护最近 5 条消息的 sender_id
        key = f"room:{room_id}:recent_senders"
        await redis_client.lpush(key, sender_id)
        await redis_client.ltrim(key, 0, MONOPOLY_THRESHOLD - 1)
        await redis_client.expire(key, 3600)
        
        recent = await redis_client.lrange(key, 0, -1)
        
        if len(recent) == MONOPOLY_THRESHOLD and len(set(recent)) == 1:
            # 连续 5 条来自同一人

                    "room_id": room_id,
                    "agent_role": "encourager",
                    "reason": f"连续{MONOPOLY_THRESHOLD}条消息来自同一人，需要邀请其他成员参与",
                    "priority": 2,
                    "triggered_at": time.time()
                })
```

### 步骤 8：调度频率控制

在 AgentWorker 执行任务前，增加额外的冷却时间检查：

```python
async def _execute_task(self, room_id: str, task: dict):
    agent_role = task.get("agent_role")
    
    # 检查角色冷却时间（同一角色短时间内不重复发言）
    cooldown_key = f"cooldown:{room_id}:{agent_role}"
    if await redis_client.exists(cooldown_key):
        # 冷却中，重新入队（稍后重试）
        await requeue_task(room_id, task, delay_seconds=10)
        return
    
    # ... 获取分布式锁 & 执行 ...
    
    # 执行完毕后设置冷却（从 agent_settings.yaml 读取）
    cooldown = agent_settings.timing.agent_cooldown_seconds
    await redis_client.setex(cooldown_key, cooldown, "1")
    
    # 全局每小时干预次数限制
    hourly_key = f"interventions:{room_id}:{int(time.time() // 3600)}"
    count = await redis_client.incr(hourly_key)
    await redis_client.expire(hourly_key, 3600)
    
    if count > agent_settings.timing.global_intervention_limit_per_hour:
        print(f"[AgentWorker] 房间 {room_id} 本小时干预已达上限")
        return  # 不执行，但不重新入队
```
