```python
import enum
from sqlalchemy import Column, String, Text, Enum, BigInteger, ForeignKey, DateTime
from sqlalchemy.dialects.postgresql import UUID
import uuid

class SenderType(str, enum.Enum):
    student = "student"
    agent = "agent"

class MessageStatus(str, enum.Enum):
    streaming = "streaming"
    ok = "ok"
    failed = "failed"

class Message(Base):
    __tablename__ = "messages"

    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    room_id = Column(UUID(as_uuid=True), ForeignKey("rooms.id"), nullable=False)
    seq_num = Column(BigInteger, nullable=False)
    sender_type = Column(Enum(SenderType), nullable=False)
    sender_id = Column(UUID(as_uuid=True), ForeignKey("users.id"), nullable=True)
    agent_role = Column(String(50), nullable=True)
    content = Column(Text, nullable=False)
    status = Column(Enum(MessageStatus), default=MessageStatus.ok)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    __table_args__ = (
        # 复合唯一约束：同一房间内 seq_num 唯一
        UniqueConstraint("room_id", "seq_num"),
        # 高性能查询索引
        Index("idx_messages_room_seq", "room_id", seq_num.desc()),
    )
```

> **关键约束：** `seq_num` 必须通过 Redis 原子计数器 `INCR room:{room_id}:msg_seq` 生成，禁止使用 `MAX(seq_num)+1`。

**文件：** `backend/app/websocket/handlers.py`

```python
from fastapi import WebSocket, WebSocketDisconnect
from app.websocket.manager import ConnectionManager
from app.services.message_service import MessageService
from app.db.session import get_db
import asyncio, json

manager = ConnectionManager()

async def websocket_endpoint(websocket: WebSocket, room_id: str):
    await websocket.accept()
    
    # === 首帧鉴权逻辑（10秒超时）===
    try:
        auth_data = await asyncio.wait_for(
            websocket.receive_json(), timeout=10.0
        )
    except asyncio.TimeoutError:
        await websocket.close(code=4002, reason="Auth timeout")
        return
    
    if auth_data.get("type") != "auth":
        await websocket.close(code=4001, reason="First frame must be auth")
        return
    
    # 验证 JWT Token
    user = await verify_token(auth_data.get("token"))
    if not user:
        await websocket.close(code=4001, reason="Invalid token")
        return
    
    # 验证是否为房间成员
    if not await is_room_member(user.id, room_id):
        await websocket.close(code=4003, reason="Not a room member")
        return
    
    # 注册连接
    await manager.connect(websocket, room_id, user)
    
    # 广播用户加入
    await manager.broadcast_to_room(room_id, {
        "type": "room:user_join",
        "user_id": str(user.id),
        "display_name": user.display_name
    })
    
    try:
        while True:
            data = await websocket.receive_json()
            
            if data.get("type") == "chat:message":
                await handle_chat_message(data, room_id, user)
                
    except WebSocketDisconnect:
        manager.disconnect(websocket, room_id)
        await manager.broadcast_to_room(room_id, {
            "type": "room:user_leave",
            "user_id": str(user.id)
        })
```

### 步骤 2：实现 ConnectionManager

**文件：** `backend/app/websocket/manager.py`

```python
from fastapi import WebSocket
from typing import dict, list, set
import json

class ConnectionManager:
    def __init__(self):
        # room_id → set of (WebSocket, user_role)
        self._rooms: dict[str, list[dict]] = {}
        # 用于 Redis 订阅协程的引用
        self._subscribers: dict[str, asyncio.Task] = {}
    
    async def connect(self, websocket: WebSocket, room_id: str, user):
        """注册新连接，并启动该房间的 Redis 订阅（若未启动）"""
        if room_id not in self._rooms:
            self._rooms[room_id] = []
            # 启动该房间的 Redis Pub/Sub 订阅协程
            task = asyncio.create_task(
                self._redis_subscriber(room_id)
            )
            self._subscribers[room_id] = task
        
        self._rooms[room_id].append({
            "ws": websocket,
            "user_id": str(user.id),
            "role": user.role.value  # "student" or "teacher"
        })
    
    def disconnect(self, websocket: WebSocket, room_id: str):
        """移除连接；若房间无人则取消订阅协程"""
        if room_id in self._rooms:
            self._rooms[room_id] = [
                c for c in self._rooms[room_id] if c["ws"] != websocket
            ]
            if not self._rooms[room_id]:
                # 房间空了，取消 Redis 订阅
                if task := self._subscribers.pop(room_id, None):
                    task.cancel()
                del self._rooms[room_id]
    
    async def broadcast_to_room(self, room_id: str, data: dict):
        """向房间内所有连接广播（通过 Redis Pub/Sub）"""
        await redis_client.publish(f"room:{room_id}", json.dumps(data))
    
    async def broadcast_to_teachers(self, room_id: str, data: dict):
        """仅向教师连接发送（analysis:update 专用）"""
        if room_id not in self._rooms:
            return
        for conn in self._rooms[room_id]:
            if conn["role"] == "teacher":
                try:
                    await conn["ws"].send_json(data)
                except Exception:
                    pass
    
    async def _redis_subscriber(self, room_id: str):
        """订阅 Redis channel，收到消息后广播给本机所有连接"""
        pubsub = redis_client.pubsub()
        await pubsub.subscribe(f"room:{room_id}")
        
        async for message in pubsub.listen():
            if message["type"] == "message":
                data = json.loads(message["data"])
                # 向本实例中该房间的所有连接发送
                if room_id in self._rooms:
                    for conn in list(self._rooms[room_id]):
                        try:
                            await conn["ws"].send_json(data)
                        except Exception:
                            pass
```

### 步骤 3：实现消息存储 Service

**文件：** `backend/app/services/message_service.py`

```python
from app.db.redis_client import redis_client
from app.models.message import Message, SenderType, MessageStatus
import uuid

class MessageService:
    
    @staticmethod
    async def get_next_seq_num(room_id: str) -> int:
        """通过 Redis 原子计数器获取单调递增的消息序号"""
        return await redis_client.incr(f"room:{room_id}:msg_seq")
    
    @staticmethod
    async def save_student_message(
        db: AsyncSession,
        room_id: str,
        user_id: str,
        content: str,
        mentions: list[str] | None = None
    ) -> Message:
        seq_num = await MessageService.get_next_seq_num(room_id)
        
        msg = Message(
            id=uuid.uuid4(),
            room_id=room_id,
            seq_num=seq_num,
            sender_type=SenderType.student,
            sender_id=user_id,
            content=content,
            status=MessageStatus.ok
        )
        db.add(msg)
        await db.commit()
        await db.refresh(msg)
        return msg
    
    @staticmethod
    async def get_history_messages(
        db: AsyncSession,
        room_id: str,
        before_seq: int | None = None,
        limit: int = 50
    ) -> dict:
        """游标分页：返回 before_seq 之前的 limit 条消息（升序）"""
        query = select(Message).where(Message.room_id == room_id)
        if before_seq:
            query = query.where(Message.seq_num < before_seq)
        query = query.order_by(Message.seq_num.desc()).limit(limit)
        
        result = await db.execute(query)
        messages = result.scalars().all()
        messages.reverse()  # 转为升序
        
        has_more = len(messages) == limit
        oldest_seq = messages[0].seq_num if messages else None
        
        return {
            "messages": messages,
            "has_more": has_more,
            "oldest_seq": oldest_seq
        }
```

### 步骤 7：前端 useWebSocket Composable

**文件：** `frontend/src/composables/useWebSocket.js`

```javascript
import { ref, onUnmounted } from 'vue'
import ReconnectingWebSocket from 'reconnecting-websocket'
import { useAuthStore } from '@/stores/auth'

export function useWebSocket(roomId) {
  const authStore = useAuthStore()
  const ws = ref(null)
  const connected = ref(false)
  const messageHandlers = new Map()
  
  function connect() {
    const wsUrl = `ws://localhost:8000/ws/${roomId}`
    ws.value = new ReconnectingWebSocket(wsUrl, [], {
      maxRetries: 10,
      reconnectionDelayGrowFactor: 1.5,
    })
    
    ws.value.onopen = () => {
      connected.value = true
      // 连接成功后立即发送首帧鉴权
      ws.value.send(JSON.stringify({
        type: 'auth',
        token: authStore.token
      }))
    }
    
    ws.value.onclose = (event) => {
      connected.value = false
      // code 4001/4002/4003 为服务端主动拒绝，不重连
      if ([4001, 4002, 4003].includes(event.code)) {
        ws.value.close()
      }
    }
    
    ws.value.onmessage = (event) => {
      const data = JSON.parse(event.data)
      const handler = messageHandlers.get(data.type)
      if (handler) handler(data)
    }
  }
  
  function on(eventType, handler) {
    messageHandlers.set(eventType, handler)
  }
  
  function send(data) {
    if (connected.value) {
      ws.value.send(JSON.stringify(data))
    }
  }
  
  function disconnect() {
    ws.value?.close()
  }
  
  onUnmounted(disconnect)
  
  return { connect, on, send, disconnect, connected }
}
```

### 步骤 8：前端 Chat Store

**文件：** `frontend/src/stores/chat.js`

```javascript
import { defineStore } from 'pinia'
import { roomApi } from '@/services/roomApi'

export const useChatStore = defineStore('chat', {
  state: () => ({
    messages: [],        // 当前房间消息列表（升序）
    hasMore: true,       // 是否还有更多历史消息
    oldestSeq: null,     // 最早消息的 seq_num（用于上拉加载更多）
    loading: false,
  }),
  
  actions: {
    // 初始化：加载最近 50 条历史消息
    async loadHistory(roomId) {
      this.loading = true
      const result = await roomApi.getMessages(roomId)
      this.messages = result.messages
      this.hasMore = result.has_more
      this.oldestSeq = result.oldest_seq
      this.loading = false
    },
    
    // 上拉加载更多（无限滚动）
    async loadMore(roomId) {
      if (!this.hasMore || this.loading) return
      this.loading = true
      const result = await roomApi.getMessages(roomId, {
        before_seq: this.oldestSeq
      })
      this.messages = [...result.messages, ...this.messages]
      this.hasMore = result.has_more
      this.oldestSeq = result.oldest_seq || this.oldestSeq
      this.loading = false
    },
    
    // 新消息追加（WebSocket 推送）
    addMessage(message) {
      // 防重复：同一 id 不重复添加
      if (!this.messages.find(m => m.id === message.id)) {
        this.messages.push(message)
      }
    },
    
    clearMessages() {
      this.messages = []
      this.hasMore = true
      this.oldestSeq = null
    }
  }
})
```

## 五、WebSocket 事件协议（P2 使用的事件）

| 事件名 | 方向 | 数据结构 | 说明 |
|--------|------|----------|------|
| `auth` | 客→服 | `{type: "auth", token: string}` | 连接后第一帧鉴权 |
| `chat:message` | 客→服 | `{type: "chat:message", content: string, mentions: []}` | 发送消息 |
| `chat:new_message` | 服→客 | `{type, id, seq_num, sender_type, sender_id, display_name, content, created_at}` | 广播新消息 |
| `room:user_join` | 服→客 | `{type, user_id, display_name}` | 用户加入广播 |
| `room:user_leave` | 服→客 | `{type, user_id}` | 用户离开广播 |

## 六、关键技术细节

### 6.1 seq_num 生成流程

```
学生发消息
  → INCR room:{room_id}:msg_seq   （Redis 原子操作，返回唯一递增数字）
  → 写入 messages 表（seq_num = Redis 返回值）
  → Redis Pub/Sub 广播（携带 seq_num）
  → 前端按 seq_num 渲染消息顺序
```

### 6.2 Redis Pub/Sub vs 本地广播

本阶段为**单进程**部署，理论上可以直接本地广播。但设计上必须通过 Redis Pub/Sub 中转，原因：

1. 为 P8 阶段多实例横向扩展提前铺路
2. AgentWorker（P3+）产生的 AI 消息也通过 Redis Pub/Sub 广播，保持统一路径

### 6.3 游标分页 vs 页码分页

使用 `seq_num` 游标分页而非页码，是为了解决：
- 实时消息插入导致翻页时消息重复出现的问题
- 分页边界稳定：`before_seq` 作为锚点，不受新消息影响

### 6.4 断线重连策略

```javascript
// reconnecting-websocket 配置
{
  maxRetries: 10,                    // 最多重连 10 次
  minReconnectionDelay: 1000,        // 最短间隔 1 秒
  maxReconnectionDelay: 30000,       // 最长间隔 30 秒
  reconnectionDelayGrowFactor: 1.5,  // 指数退避
}
```

重连成功后前端自动重新发送 `auth` 首帧（`onopen` 回调中执行）。
