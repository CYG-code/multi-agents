from collections import Counter
from app.models.message import Message

def calculate_participation_scores(messages: list[dict]) -> dict:
    """
    计算各成员参与度得分（发言次数归一化）。
    返回 {user_id: score} 格式。
    """
    student_msgs = [m for m in messages if m.get("sender_type") == "student"]
    counts = Counter(m.get("sender_id") for m in student_msgs)
    
    if not counts:
        return {}
    
    total = sum(counts.values())
    return {uid: count / total for uid, count in counts.items()}


def calculate_balance_score(participation_scores: dict) -> float:
    """
    协作均衡性 = 1 - 基尼系数（越均衡越接近1）。
    """
    if not participation_scores:
        return 0.5
    
    values = sorted(participation_scores.values())
    n = len(values)
    
    # 基尼系数计算
    cumulative = sum((i + 1) * v for i, v in enumerate(values))
    total = sum(values)
    
    if total == 0:
        return 0.5
    
    gini = (2 * cumulative) / (n * total) - (n + 1) / n
    return max(0.0, min(1.0 - abs(gini), 1.0))


def extract_message_stats(messages: list[dict]) -> dict:
    """提取消息统计信息，供 Analyst 使用"""
    student_msgs = [m for m in messages if m.get("sender_type") == "student"]
    
    return {
        "total_count": len(messages),
        "student_count": len(student_msgs),
        "agent_count": len(messages) - len(student_msgs),
        "unique_participants": len(set(m.get("sender_id") for m in student_msgs if m.get("sender_id"))),
        "avg_message_length": sum(len(m.get("content", "")) for m in student_msgs) / max(len(student_msgs), 1),
    }
```

### 步骤 3：实现三位分析专家（AI 版本）

**文件：** `backend/app/background_experts/cognitive_analyst.py`

```python
import json
from app.agents.claude_client import get_claude_client
from app.config import agent_settings
from app.analysis.metrics import extract_message_stats

class CognitiveAnalyst:
    MODEL = None  # 从 agent_settings 读取
    
    def __init__(self):
        with open("app/agents/prompts/analysts/cognitive_analyst_prompt.txt") as f:
            self._prompt_template = f.read()
        self.MODEL = agent_settings.models.cognitive_analyst.model_version
        self.TOKEN_BUDGET = agent_settings.models.cognitive_analyst.history_token_budget
    
    async def analyze(
        self,
        messages: list[dict],
        task: dict,
        current_phase: str
    ) -> dict:
        """
        并发调用入口。返回结构化分析结果。
        """
        stats = extract_message_stats(messages)
        chat_history = self._format_history(messages)
        
        system_prompt = self._prompt_template.format(
            task_requirements=task.get("requirements", ""),
            current_phase=current_phase,
            participant_stats=json.dumps(stats, ensure_ascii=False),
            message_count=len(messages),
            chat_history=chat_history
        )
        
        client = get_claude_client()
        
        try:
            response = await client.messages.create(
                model=self.MODEL,
                max_tokens=1024,
                system=system_prompt,
                messages=[{"role": "user", "content": "请输出分析结果 JSON。"}]
            )
            
            raw = response.content[0].text.strip()
            # 去除可能的 Markdown 代码块包裹
            if raw.startswith("```"):
                raw = raw.split("```")[1]
                if raw.startswith("json"):
                    raw = raw[4:]
            
            result = json.loads(raw)
            return result
        
        except Exception as e:
            print(f"[CognitiveAnalyst] 分析失败: {e}")
            # 降级：返回默认值
            return {
                "diversity_score": 0.5,
                "progress_score": 0.5,
                "cognitive_report": {"diagnosis": "分析暂时不可用", "error": str(e)}
            }
    
    def _format_history(self, messages: list[dict]) -> str:
        """格式化聊天记录（按 Token Budget 截断）"""
        lines = []
        total_chars = 0
        # 每个 token ≈ 1.5 中文字符，预估
        char_budget = self.TOKEN_BUDGET * 1
        
        for msg in reversed(messages):  # 从最新消息开始
            line = f"[{msg.get('display_name', '?')}] {msg.get('content', '')}"
            if total_chars + len(line) > char_budget:
                lines.insert(0, "（以下为按 token budget 截断后的近期消息）")
                break
            lines.insert(0, line)
            total_chars += len(line)
        
        return "\n".join(lines)
```

class ChiefDispatcher:

            self._prompt_template = f.read()

        self,
        cognitive_report: dict,
        emotional_report: dict,
        interaction_report: dict,
        current_phase: str,
        elapsed_minutes: float
    ) -> dict:
        """
        接收三份专家报告，输出调度决策。
        注意：ChiefDispatcher 不接收原始聊天记录。
        """

        )
        
        client = get_claude_client()
        
        try:
            response = await client.messages.create(
                model=self.MODEL,
                max_tokens=512,
                system=system_prompt,
                messages=[{"role": "user", "content": "请输出调度决策 JSON。"}]
            )
            
            raw = response.content[0].text.strip()
            if raw.startswith("```"):
                raw = raw.split("```")[1]
                if raw.startswith("json"):
                    raw = raw[4:]
            
            return json.loads(raw)
        
        except Exception as e:
            print(f"[ChiefDispatcher] 调度失败: {e}")
            return {
                "dispatcher_summary": "调度分析暂时不可用",
                "interventions": []

            cognitive_analyst.analyze(messages, task_info, current_phase),

        balance = calculate_balance_score(participation)

                balance_score=balance,
                interaction_report=interaction_result.get("interaction_report", {}),

            db.add(snapshot)

            cognitive_report=cognitive_result.get("cognitive_report", {}),
            emotional_report=emotional_result.get("emotional_report", {}),
            interaction_report=interaction_result.get("interaction_report", {}),
            current_phase=current_phase,

                "room_id": room_id,

            })

    await manager.broadcast_to_room(room_id, {...})  # 广播消息
    
    # A 类检测
    await trigger_detector.check_monopoly(room_id, str(user.id))
    
    # C 类触发：@Agent 主动召唤
    if mentions and agent_settings.mention.enabled:
        valid_roles = list(ROLE_AGENTS.keys())
        for role in mentions[:agent_settings.mention.max_mentions_per_message]:
            if role not in valid_roles:
                continue
            
            # 写入 agent_interventions（C 类，snapshot_id=null）
            intervention = AgentIntervention(
                room_id=room_id,
                snapshot_id=None,  # C 类不关联快照
                agent_role=role,
                reason=f"学生 {user.display_name} 通过 @ 主动召唤：{content[:100]}"
            )
            db.add(intervention)
            await db.commit()
            
            # 写入 priority=0 的队列任务
            await enqueue_task(room_id, {
                "room_id": room_id,
                "agent_role": role,
                "reason": content,
                "priority": agent_settings.mention.priority,
                "trigger_type": "mention",
                "student_name": user.display_name,
                "triggered_at": time.time()
            })
```

from fastapi import APIRouter, Depends
from app.models.analysis import AnalysisSnapshot, AgentIntervention

analytics_router = APIRouter(prefix="/api/analytics", tags=["分析"])

@analytics_router.get("/rooms/{room_id}/snapshots")
async def get_snapshots(
    room_id: UUID,
    limit: int = Query(20, le=50),
    current_user: User = Depends(require_teacher),
    db: AsyncSession = Depends(get_db)
):
    """获取协作分析快照列表（教师端专用）"""
    result = await db.execute(
        select(AnalysisSnapshot)
        .where(AnalysisSnapshot.room_id == room_id)
        .order_by(AnalysisSnapshot.created_at.desc())
        .limit(limit)
    )
    snapshots = result.scalars().all()
    return {"snapshots": snapshots}


@analytics_router.get("/rooms/{room_id}/interventions")
async def get_interventions(
    room_id: UUID,
    limit: int = Query(50, le=100),
    current_user: User = Depends(require_teacher),
    db: AsyncSession = Depends(get_db)
):
    """获取干预记录列表（教师端专用）"""
    result = await db.execute(
        select(AgentIntervention)
        .where(AgentIntervention.room_id == room_id)
        .order_by(AgentIntervention.created_at.desc())
        .limit(limit)
    )
    interventions = result.scalars().all()
    return {"interventions": interventions}


@analytics_router.get("/rooms/{room_id}/summary")
async def get_summary(
    room_id: UUID,
    current_user: User = Depends(require_teacher),
    db: AsyncSession = Depends(get_db)
):
    """获取整体协作摘要（最新快照的综合指标）"""
    latest = await db.execute(
        select(AnalysisSnapshot)
        .where(AnalysisSnapshot.room_id == room_id)
        .order_by(AnalysisSnapshot.created_at.desc())
        .limit(1)
    )
    snapshot = latest.scalar_one_or_none()
    
    if not snapshot:
        return {"message": "暂无分析数据"}
    
    return {
        "diversity_score": snapshot.diversity_score,
        "balance_score": snapshot.balance_score,
        "progress_score": snapshot.progress_score,
        "emotion_flags": snapshot.emotion_flags,
        "dispatcher_summary": snapshot.dispatcher_summary,
        "last_analysis_at": snapshot.created_at.isoformat()
    }
